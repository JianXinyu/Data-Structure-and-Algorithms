# Solution 1
求和再减
```cpp
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        const int n = nums.size();
        int sum = n*(n+1) / 2;
        for (int n : nums) {
            sum -= n;
        }
        return sum;
    }
	
	int missingNumber(vector<int>& nums) {
        const int n = nums.size();
        int ans = 0;
        for (int i = 1; i <= n; ++i) {
            ans += i - nums[i-1];
        }
        return ans;
    }
};
```

# Solution 2
#bit_operation

```cpp
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int res = 0;
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            res ^= nums[i];
        }
        for (int i = 0; i <= n; i++) {
            res ^= i;
        }
        return res;
    }
};
```
