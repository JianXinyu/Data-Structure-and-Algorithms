class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        if(!head->next || left == right) 
            return head;
        
        ListNode dummy(0);
        dummy.next = head;
        ListNode *p = &dummy;
        // Find the m-1 node
        for(int i = 0; i < left-1; ++i){
            p = p->next;
        }
        // Process node m
        ListNode *prev = p;
        ListNode *cur = p->next;
        ListNode *tail = cur;
        // Reverse from m to n
        for(int i = left; i <= right; ++i){
            ListNode *next = cur->next;
            cur->next = prev;
            prev = cur;
            cur = next;
        }
        // connect three parts
        p->next = prev;
        tail->next = cur;
        return dummy.next;
    }
};
