int largestRectangleArea(vector<int>& heights) {
	heights.push_back(0);
	const int n = heights.size();
	stack<int> s;
	int ans = 0;
	for(int i = 0; i < n; ++i){
		while(!s.empty() && heights[s.top()] > heights[i]){
			int tmp = heights[s.top()];
			s.pop();
			// corner case
			int j = s.size() ? s.top() : -1;
			ans = max(ans, tmp*(i - j - 1));
		}
		s.push(i);
	}
	return ans;
}
