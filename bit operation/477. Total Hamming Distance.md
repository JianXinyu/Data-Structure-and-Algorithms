```cpp
int totalHammingDistance(vector<int>& nums) {
	int ans = 0;
	int64_t mask = 1;
	for (int i = 0; i < 32; ++i) {
		int count = 0;
		for (const int num : nums)
			if (num & mask) ++count;
		ans += (nums.size() - count) * count;
		mask <<= 1;
	}
	return ans;
}
```
T: O(n)
S: O(1)
