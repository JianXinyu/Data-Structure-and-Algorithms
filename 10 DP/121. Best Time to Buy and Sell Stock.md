```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        const int n = prices.size();
        if (n == 1)
            return 0;
        vector<int> dp(n, 0);
        int m = prices[0];
        for (int i = 1; i < n; ++i) {
            dp[i] = max(dp[i-1], prices[i] - m);
            // printf("dp[%d]: %d, m: %d\n", i, dp[i], m);
            m = min(prices[i], m);
        }
        return dp[n-1];
    }
};
```
T: O(n)
S: O(n)

改进空间复杂度 S: O(1)
```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        const int n = prices.size();
        if (n == 1)
            return 0;
        int dp = 0;
        int m = prices[0];
        for (int i = 1; i < n; ++i) {
            dp = max(dp, prices[i] - m);
            m = min(prices[i], m);
        }
        return dp;
    }
};
```
